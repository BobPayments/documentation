# Listar Transações

Retorna uma lista paginada das transações do projeto, com suporte a filtros por status, período, valor e busca textual.

## Query Parameters

<ParamField query="page" type="integer">
  Número da página. Default: `1`.
</ParamField>
<ParamField query="limit" type="integer">
  Itens por página (máximo 100). Default: `10`.
</ParamField>
<ParamField query="status" type="string">
  Filtrar por status. Valores: `waiting_payment`, `processing`, `paid`, `expired`, `cancelled`, `refunded`.
</ParamField>
<ParamField query="customerId" type="string">
  Filtrar por ID do cliente (formato CUID).
</ParamField>
<ParamField query="dateFrom" type="string">
  Data mínima de criação (ISO 8601).
</ParamField>
<ParamField query="dateTo" type="string">
  Data máxima de criação (ISO 8601).
</ParamField>
<ParamField query="paidAtFrom" type="string">
  Data mínima de pagamento (ISO 8601).
</ParamField>
<ParamField query="paidAtTo" type="string">
  Data máxima de pagamento (ISO 8601).
</ParamField>
<ParamField query="expiredAtFrom" type="string">
  Data mínima de expiração (ISO 8601).
</ParamField>
<ParamField query="expiredAtTo" type="string">
  Data máxima de expiração (ISO 8601).
</ParamField>
<ParamField query="amountMin" type="integer">
  Valor mínimo em centavos.
</ParamField>
<ParamField query="amountMax" type="integer">
  Valor máximo em centavos.
</ParamField>
<ParamField query="search" type="string">
  Busca textual por nome ou email do cliente. Máximo 200 caracteres.
</ParamField>
<ParamField query="environment" type="string">
  Filtrar por ambiente: `live` ou `sandbox`.
</ParamField>
<ParamField query="gatewayId" type="string">
  Filtrar por gateway específico (formato CUID).
</ParamField>
<ParamField query="originDomain" type="string">
  Filtrar pelo domínio exato de origem da transação.
</ParamField>
<ParamField query="domain" type="string">
  Busca por domínio em origem ou links de pagamento associados.
</ParamField>

<ResponseField name="data[].customer" type="object">
  Dados resumidos do cliente (id, name, email).
</ResponseField>
<ResponseField name="data[].gateway" type="object">
  Gateway que processou a transação (id, alias, providerId, providerName).
</ResponseField>
<ResponseField name="data[].originDomain" type="string | null">
  Domínio de origem da transação. `null` se não houve domínio registrado.
</ResponseField>
<ResponseField name="meta" type="object">
  Metadados de paginação com total de registros, página atual, limite e total de páginas.
</ResponseField>
